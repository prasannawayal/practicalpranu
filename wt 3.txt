String_function.php 

 

<?php 

function stringLength($str){ 

$count=0; 

while(isset($str[$count])){ 

$count++; 

} 

return $count; 

} 

 function countvowels($str){ 

 $vowels=['a','e','i','o','u']; 

 $count=0; 

 $str=strtolower($str); 

 for($i=0;$i<strlen($str);$i++){ 

 if (in_array($str[$i],$vowels)){ 

 $count++; 

 } 

 } 

 return $count; 



 } 

function convertTitleCase($str) 

{ 

  $str=strtolower($str); 

  return ucwords($str); 

  }   

function padString($str){ 

   return str_pad($str,strlen($str)+4,"*",STR_PAD_BOTH); 

   }    

function trimLeadingSpaces($str){ 

  return ltrim($str); 

}    

function reverseString($str) 

{ 

  return strrev($str); 

} 

?>
